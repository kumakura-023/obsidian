```C#
public static class HighSchoolSweethearts
{
	//61文字のど真ん中にハートを配置し、その左右に名前を格納
    public static string DisplaySingleLine(string name1, string name2)
    {
        const int width = 61;
        const char heart = '♡';

        string text = $"{name1} {heart} {name2}";

        int targetHeartIndex = width / 2; // 61幅の真ん中 → 30
        int heartIndexInText = text.IndexOf(heart);

        // ハートを中央に持っていくための左側スペース数
        int leftPadding = targetHeartIndex - heartIndexInText;

        // 左パディング → 長さ揃えるための右パディング
        string padded = text.PadLeft(leftPadding + text.Length)
                             .PadRight(width);

        return padded;
    }
    ```

### 日付、小数のフォーマット形式設定
```C#
    //日付、小数のフォーマット形式設定
    public static string DisplayGermanExchangeStudents(
    string studentA, string studentB, DateTime start, float hours)
    {
        // ドイツ文化で日付をフォーマット
        var german = new System.Globalization.CultureInfo("de-DE");
    
        string date = start.ToString("dd.MM.yyyy", german);
        string hrs  = hours.ToString("N2", german); // 小数点3桁
    
        return $"{studentA} and {studentB} have been dating since {date} - that's {hrs} hours";
    }
}
```
