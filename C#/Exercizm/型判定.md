public static string AnalyzeOffField(object report)
{
    if (report is int count)
    {
        return $"There are {count} supporters at the match.";
    }
    else if (report is string s)
    {
        return s;
    }
    else if (report is Incident incident)
    {
        return incident.GetDescription();
    }
    else if (report is Manager manager)
    {
        if (manager.Club == null)
        {
            return $"{manager.Name}";
        }
        else
        {
            return $"{manager.Name}({manager.Club})";
        }
    }
    else return "";
}
